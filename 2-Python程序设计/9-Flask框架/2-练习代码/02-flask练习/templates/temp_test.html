常见内建filter

hello

{{ 'hello' }}

{# 自动转义 #}

{{ 'hello'|safe }}

{# 变量首字母大写 #}

{{ 'hello'|capitalize }}

{# 转化成大写 #}

{{ 'hello'|upper }}

{# 转化成小写 #}

{{ 'HELLO'|lower }}

{# 单词首字母大写 #}

{{ 'ada addada fhfghf'|title }}

{# 格式化输出 #}

{{ '%s is %d'| format('李傲',18) }}

{# 反转 #}

{{ 'hello'|reverse }}

{# 渲染之前把所有html标签都去掉 #}

{{ '

hello

'|striptags }}

列表操作

{# 取值 #}

{{ [1,2,3,4 ]|first }}

{{ [1,2,3,4 ]|last }}

{{ [1,2,3,4 ]|length }}

{# 求和 #}

{{ [1,2,3,4 ]|sum }}

{# 排序 #}

{{ [1,2,3,4 ]|sort }}
